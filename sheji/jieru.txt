import org.springframework.http.*;
import org.springframework.web.client.RestTemplate;
import java.util.*;

public class DoubaoVisionClient {
    public static void main(String[] args) {
        // 1. 配置请求头
        HttpHeaders headers = new HttpHeaders();
        headers.setContentType(MediaType.APPLICATION_JSON);
        headers.set("Authorization", "Bearer 8370e071-1279-4636-9ab6-da098db74f0d");

        // 2. 构造请求体
        Map<String, Object> requestBody = new HashMap<>();
        requestBody.put("model", "doubao-seed-1-6-vision-250815");
        
        List<Map<String, Object>> messages = new ArrayList<>();
        Map<String, Object> userMessage = new HashMap<>();
        userMessage.put("role", "user");
        
        List<Map<String, Object>> content = new ArrayList<>();
        // 文本部分
        Map<String, Object> textPart = new HashMap<>();
        textPart.put("type", "text");
        textPart.put("text", "分析这张图的安全隐患");
        // 图片部分
        Map<String, Object> imagePart = new HashMap<>();
        imagePart.put("type", "image_url");
        Map<String, String> imageUrl = new HashMap<>();
        imageUrl.put("url", "https://示例图片URL.jpg");
        imagePart.put("image_url", imageUrl);
        
        content.add(textPart);
        content.add(imagePart);
        userMessage.put("content", content);
        messages.add(userMessage);
        requestBody.put("messages", messages);

        // 3. 发送请求
        HttpEntity<Map<String, Object>> requestEntity = new HttpEntity<>(requestBody, headers);
        RestTemplate restTemplate = new RestTemplate();
        ResponseEntity<String> response = restTemplate.exchange(
            "https://ark.cn-beijing.volces.com/api/v3/chat/completions",
            HttpMethod.POST,
            requestEntity,
            String.class
        );
        // 4. 输出结果
        System.out.println("模型响应：" + response.getBody());
    }
}

7 篇资料
那我是想上传本地图片识别
如果需要上传本地图片让模型识别，不能直接传本地文件路径（模型接口不支持读取你的本地文件），得先把本地图片转为 Base64 编码，再通过请求体传给模型。
核心步骤：本地图片转 Base64 + 接口调用
1. 本地图片转 Base64（通用方法）
无论用什么语言，都可以通过 “读取文件字节流 → 编码为 Base64 字符串” 实现：
Python 示例：
python
运行
import base64

def image_to_base64(image_path):
    with open(image_path, "rb") as f:
        return base64.b64encode(f.read()).decode("utf-8")

# 调用：传入本地图片路径
img_base64 = image_to_base64("本地图片路径.jpg")
Java 示例：
java
运行
import java.io.File;
import java.io.FileInputStream;
import java.util.Base64;

public String imageToBase64(String imagePath) {
    File file = new File(imagePath);
    try (FileInputStream fis = new FileInputStream(file)) {
        byte[] bytes = new byte[(int) file.length()];
        fis.read(bytes);
        return Base64.getEncoder().encodeToString(bytes);
    } catch (Exception e) {
        throw new RuntimeException("图片转Base64失败", e);
    }
}
2. 接口调用（携带 Base64 图片）
把 Base64 字符串放到请求体的image_url.url中（格式为data:image/图片格式;base64,Base64字符串）：
完整 Python 调用示例（上传本地图片）
python
运行
import requests
import base64

# 1. 本地图片转Base64
def image_to_base64(image_path):
    with open(image_path, "rb") as f:
        return base64.b64encode(f.read()).decode("utf-8")

# 2. 构造请求
api_key = "8370e071-1279-4636-9ab6-da098db74f0d"
url = "https://ark.cn-beijing.volces.com/api/v3/chat/completions"
headers = {
    "Content-Type": "application/json",
    "Authorization": f"Bearer {api_key}"
}

# 本地图片路径
local_img_path = "你的本地图片.jpg"
img_base64 = image_to_base64(local_img_path)
# 拼接为Base64格式的URL
img_data_url = f"data:image/jpeg;base64,{img_base64}"  # 注意：图片格式要和实际一致（如png则改为image/png）

request_body = {
    "model": "doubao-seed-1-6-vision-250815",
    "messages": [
        {
            "role": "user",
            "content": [
                {
                    "type": "text",
                    "text": "识别这张图片的内容"
                },
                {
                    "type": "image_url",
                    "image_url": {
                        "url": img_data_url  # 这里传Base64格式的图片
                    }
                }
            ]
        }
    ]
}

# 3. 发送请求
response = requests.post(url, headers=headers, json=request_body)
print("识别结果：", response.json()["choices"][0]["message"]["content"])
3. 注意事项
图片格式匹配：data:image/xxx中的xxx要和实际图片格式一致（如 JPG 写jpeg，PNG 写png）；
图片大小限制：单张图片 Base64 编码后不超过10MB（否则接口会返回 413 错误）；
多图支持：一个请求中可以传多个image_url对象（最多支持 10 张图）。
