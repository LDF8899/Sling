医院服务（Hospital-Service）设计方案（高德 API 专属版）
1. 模块概述
医院服务模块是 SLING 项目的重要组成部分，主要负责医院信息管理、血清库存跟踪、用户周边医院查找、蛇毒治疗专项医院筛选等核心功能。该模块与预警服务共享高德地图 API 功能，实现基于地理位置的医院搜索、地址解析、距离计算与专项医疗资源筛选服务，重点强化蛇毒咬伤治疗相关医院的精准检索能力，为紧急医疗场景提供高效支持。
2. 数据库表结构

6.1 完善 AmapService（核心地图服务）
java
运行
package com.sling.hospital.service;
import com.alibaba.fastjson2.JSON;
import com.alibaba.fastjson2.JSONObject;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;
import org.springframework.web.client.RestTemplate;
import jakarta.annotation.PostConstruct;
import java.util.HashMap;
import java.util.Map;

@Service
public class AmapService {
    @Value("${amap.api-key:}")
    private String amapApiKey;

    private RestTemplate restTemplate;

    // 高德地图API接口地址
    private static final String GEOCODE_URL = "https://restapi.amap.com/v3/geocode/geo";
    private static final String POI_SEARCH_URL = "https://restapi.amap.com/v3/place/around";

    @PostConstruct
    public void init() {
        this.restTemplate = new RestTemplate();
    }

    /**
     * 解析用户输入（支持地址或经纬度），返回经纬度（高德GCJ02坐标系）
     * @param userInput 用户输入（如"广州市天河区"或"113.32,23.13"）
     * @return 经纬度数组[lng, lat]
     */
    public Double[] parseUserInput(String userInput) throws Exception {
        if (userInput.contains(",")) {
            // 输入为经纬度格式
            String[] lngLatArr = userInput.split(",");
            if (lngLatArr.length != 2) {
                throw new IllegalArgumentException("经纬度格式错误，正确格式为\"经度,纬度\"");
            }
            Double lng = Double.parseDouble(lngLatArr[0].trim());
            Double lat = Double.parseDouble(lngLatArr[1].trim());
            return new Double[]{lng, lat};
        } else {
            // 输入为地址，转换为经纬度（高德GCJ02坐标系）
            String geoCodeUrl = String.format(
                "%s?key=%s&address=%s&output=json",
                GEOCODE_URL, amapApiKey, userInput
            );
            String response = restTemplate.getForObject(geoCodeUrl, String.class);
            JSONObject result = JSON.parseObject(response);
            if (!"1".equals(result.getString("status")) || result.getJSONArray("geocodes").isEmpty()) {
                throw new Exception("地址解析失败：" + result.getString("info"));
            }
            String lngLat = result.getJSONArray("geocodes").getJSONObject(0).getString("location");
            String[] lngLatArr = lngLat.split(",");
            Double lng = Double.parseDouble(lngLatArr[0]);
            Double lat = Double.parseDouble(lngLatArr[1]);
            return new Double[]{lng, lat};
        }
    }

    /**
     * 周边POI搜索（查找附近医院）
     * @param lng 中心点经度
     * @param lat 中心点纬度
     * @param radius 搜索半径（米）
     * @param pageSize 每页结果数
     * @return 医院原始数据JSON字符串
     */
    public String searchNearbyHospitals(Double lng, Double lat, Integer radius, Integer pageSize) throws Exception {
        Map<String, Object> params = new HashMap<>();
        params.put("key", amapApiKey);
        params.put("location", lng + "," + lat); // 高德POI搜索：经度在前，纬度在后
        params.put("radius", radius);
        params.put("keywords", "医院");
        params.put("types", "医疗保健服务;医疗服务;医院");
        params.put("output", "json");
        params.put("offset", pageSize);
        params.put("page", 1);
        params.put("extensions", "all"); // 返回详细信息

        String response = restTemplate.getForObject(POI_SEARCH_URL, String.class, params);
        if (response == null) {
            throw new Exception("高德地图POI搜索失败，无返回结果");
        }

        JSONObject result = JSON.parseObject(response);
        if (!"1".equals(result.getString("status"))) {
            throw new Exception("周边医院检索失败：" + result.getString("info"));
        }

        return response;
    }

    /**
     * 按关键词搜索医院（兜底方案专用）
     * @param keywords 搜索关键词
     * @param city 城市（可选）
     * @return 医院原始数据JSON字符串
     */
    public String searchHospitalsByKeyword(String keywords, String city) throws Exception {
        String searchUrl = String.format(
            "https://restapi.amap.com/v3/place/text?key=%s&keywords=%s&city=%s&output=json&offset=20&extensions=all",
            amapApiKey, keywords, city
        );
        String response = restTemplate.getForObject(searchUrl, String.class);
        if (response == null) {
            throw new Exception("高德地图关键词搜索失败，无返回结果");
        }

        JSONObject result = JSON.parseObject(response);
        if (!"1".equals(result.getString("status"))) {
            throw new Exception("关键词医院检索失败：" + result.getString("info"));
        }

        return response;
    }
}
6.2 蛇毒治疗医院服务（SnakeVenomHospitalService）
java
运行
package com.sling.hospital.service;
import com.alibaba.fastjson2.JSON;
import com.alibaba.fastjson2.JSONObject;
import com.sling.hospital.entity.HospitalInfo;
import com.sling.hospital.entity.dto.SnakeVenomHospitalQueryDTO;
import com.sling.hospital.mapper.HospitalInfoMapper;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import java.math.BigDecimal;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

@Service
public class SnakeVenomHospitalService {

    @Autowired
    private HospitalInfoMapper hospitalInfoMapper;

    @Autowired
    private AmapService amapService;

    // 蛇毒治疗核心关键词库（多维度覆盖）
    private static final List<String> SNAKE_VENOM_KEYWORDS = Arrays.asList(
        "蛇毒", "抗蛇毒血清", "蛇咬伤", "蛇伤治疗", "毒蛇咬伤",
        "蛇伤专科", "急诊科 蛇伤", "中毒救治 蛇", "蛇毒抗体", "蛇伤急救中心"
    );

    /**
     * 检索周边支持蛇毒治疗的医院（核心功能）
     * @param queryDTO 查询参数（用户输入、搜索半径、地图类型）
     * @return 筛选后的医院列表（含距离信息、匹配关键词）
     */
    public List<HospitalInfo> searchSnakeVenomHospitals(SnakeVenomHospitalQueryDTO queryDTO) throws Exception {
        // 1. 解析用户输入，获取经纬度（仅使用高德地图）
        Double[] lngLat = amapService.parseUserInput(queryDTO.getUserInput().trim());
        Double userLng = lngLat[0];
        Double userLat = lngLat[1];

        // 2. 查找附近支持蛇毒治疗的医院
        List<HospitalInfo> nearbyHospitals = hospitalInfoMapper.selectNearbySnakeVenomHospitals(
            new BigDecimal(userLat),
            new BigDecimal(userLng),
            queryDTO.getRadius(),
            SNAKE_VENOM_KEYWORDS
        );

        // 3. 补充距离描述和匹配关键词
        for (HospitalInfo hospital : nearbyHospitals) {
            // 计算距离描述
            double distance = hospital.getDistance();
            if (distance >= 1000) {
                hospital.setDistanceDesc(String.format("%.1f公里", distance / 1000));
            } else {
                hospital.setDistanceDesc(String.format("%.0f米", distance));
            }

            // 匹配关键词筛选
            List<String> matchedKeywords = new ArrayList<>();
            String keywords = hospital.getSnakeVenomKeywords() != null ? hospital.getSnakeVenomKeywords() : "";
            String name = hospital.getHospitalName() != null ? hospital.getHospitalName() : "";
            for (String keyword : SNAKE_VENOM_KEYWORDS) {
                if (keywords.contains(keyword) || name.contains(keyword)) {
                    matchedKeywords.add(keyword);
                }
            }
            hospital.setMatchedSnakeKeywords(matchedKeywords);
        }

        // 4. 若数据库中无结果，调用高德地图API补充检索（兜底方案）
        if (nearbyHospitals.isEmpty()) {
            nearbyHospitals = supplementHospitalsFromAmapAPI(queryDTO, userLng, userLat);
        }

        return nearbyHospitals;
    }

    /**
     * 从高德地图API补充检索医院（数据库无结果时兜底）
     */
    private List<HospitalInfo> supplementHospitalsFromAmapAPI(SnakeVenomHospitalQueryDTO queryDTO, Double userLng, Double userLat) throws Exception {
        List<HospitalInfo> supplementHospitals = new ArrayList<>();
        // 调用高德地图周边搜索接口
        String mapResult = amapService.searchNearbyHospitals(userLng, userLat, queryDTO.getRadius(), 20);

        // 解析高德地图API返回结果，转换为HospitalInfo对象
        JSONObject resultObj = JSON.parseObject(mapResult);
        if (resultObj.containsKey("pois") && !resultObj.getJSONArray("pois").isEmpty()) {
            for (Object obj : resultObj.getJSONArray("pois")) {
                JSONObject hospitalObj = (JSONObject) obj;
                HospitalInfo hospital = new HospitalInfo();

                // 填充核心字段
                hospital.setHospitalName(hospitalObj.getString("name"));
                hospital.setAddress(hospitalObj.getString("address"));
                hospital.setContactInfo(hospitalObj.getString("tel", "暂无"));
                hospital.setHospitalType(hospitalObj.getString("type"));

                // 解析经纬度
                String location = hospitalObj.getString("location");
                String[] lngLatArr = location.split(",");
                hospital.setLongitude(new BigDecimal(lngLatArr[0]));
                hospital.setLatitude(new BigDecimal(lngLatArr[1]));

                // 计算距离
                double distance = calculateDistance(
                    userLng, userLat,
                    Double.parseDouble(lngLatArr[0]), Double.parseDouble(lngLatArr[1])
                );
                hospital.setDistance(distance);
                hospital.setDistanceDesc(distance >= 1000 ?
                    String.format("%.1f公里", distance / 1000) :
                    String.format("%.0f米", distance));

                // 匹配蛇毒关键词
                List<String> matchedKeywords = new ArrayList<>();
                String name = hospital.getHospitalName() != null ? hospital.getHospitalName() : "";
                String snippet = hospitalObj.getString("snippet", "");
                String type = hospital.getHospitalType() != null ? hospital.getHospitalType() : "";
                for (String keyword : SNAKE_VENOM_KEYWORDS) {
                    if (name.contains(keyword) || snippet.contains(keyword) || type.contains(keyword)) {
                        matchedKeywords.add(keyword);
                    }
                }
                hospital.setMatchedSnakeKeywords(matchedKeywords);
                hospital.setSnakeVenomTreatment(!matchedKeywords.isEmpty());
                // 默认为有急诊科和血清（实际需用户自行确认）
                hospital.setEmergencyDepartment(true);
                hospital.setSerumAvailability(!matchedKeywords.isEmpty());

                supplementHospitals.add(hospital);
            }

            // 按距离排序
            supplementHospitals.sort((h1, h2) -> h1.getDistance().compareTo(h2.getDistance()));
        }

        return supplementHospitals;
    }

    /**
     * Haversine公式计算两点球面距离（单位：米）
     */
    private double calculateDistance(double lng1, double lat1, double lng2, double lat2) {
        final double R = 6371000; // 地球半径（米）
        double lat1Rad = Math.toRadians(lat1);
        double lat2Rad = Math.toRadians(lat2);
        double deltaLat = Math.toRadians(lat2 - lat1);
        double deltaLng = Math.toRadians(lng2 - lng1);

        double a = Math.sin(deltaLat / 2) * Math.sin(deltaLat / 2) +
                   Math.cos(lat1Rad) * Math.cos(lat2Rad) *
                   Math.sin(deltaLng / 2) * Math.sin(deltaLng / 2);
        double c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));

        return R * c;
    }

    /**
     * 导出蛇毒治疗医院信息到Excel
     * @param hospitals 医院列表
     * @return Excel文件字节数组
     */
    public byte[] exportSnakeVenomHospitalsToExcel(List<HospitalInfo> hospitals) throws Exception {
        // 使用Apache POI实现Excel导出逻辑
        org.apache.poi.ss.usermodel.Workbook workbook = new org.apache.poi.xssf.usermodel.XSSFWorkbook();
        org.apache.poi.ss.usermodel.Sheet sheet = workbook.createSheet("蛇毒治疗医院列表");

        // 创建表头
        String[] headers = {"序号", "医院名称", "地址", "联系电话", "距离", "医院类型", "是否有血清", "匹配关键词"};
        org.apache.poi.ss.usermodel.Row headerRow = sheet.createRow(0);
        for (int i = 0; i < headers.length; i++) {
            org.apache.poi.ss.usermodel.Cell cell = headerRow.createCell(i);
            cell.setCellValue(headers[i]);
        }

        // 填充数据
        for (int i = 0; i < hospitals.size(); i++) {
            HospitalInfo hospital = hospitals.get(i);
            org.apache.poi.ss.usermodel.Row dataRow = sheet.createRow(i + 1);

            dataRow.createCell(0).setCellValue(i + 1);
            dataRow.createCell(1).setCellValue(hospital.getHospitalName());
            dataRow.createCell(2).setCellValue(hospital.getAddress());
            dataRow.createCell(3).setCellValue(hospital.getContactInfo());
            dataRow.createCell(4).setCellValue(hospital.getDistanceDesc());
            dataRow.createCell(5).setCellValue(hospital.getHospitalType());
            dataRow.createCell(6).setCellValue(hospital.getSerumAvailability() ? "是" : "否");
            dataRow.createCell(7).setCellValue(String.join(",", hospital.getMatchedSnakeKeywords()));
        }

        // 自动调整列宽
        for (int i = 0; i < headers.length; i++) {
            sheet.autoSizeColumn(i);
        }

        // 写入字节数组
        java.io.ByteArrayOutputStream outputStream = new java.io.ByteArrayOutputStream();
        workbook.write(outputStream);
        workbook.close();

        return outputStream.toByteArray();
    }
}

package com.sling.hospital.controller;
import com.sling.common.utils.Result;
import com.sling.common.utils.ResultUtils;
import com.sling.hospital.entity.HospitalInfo;
import com.sling.hospital.entity.dto.SnakeVenomHospitalQueryDTO;
import com.sling.hospital.service.HospitalInfoService;
import com.sling.hospital.service.SnakeVenomHospitalService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpHeaders;
import org.springframework.http.HttpStatus;
import org.springframework.http.MediaType;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;
import javax.validation.Valid;
import java.util.List;

@RestController
@RequestMapping("/hospital")
@CrossOrigin(origins = "*")
public class HospitalInfoController {

    // 原有Autowired保持不变
    @Autowired
    private SnakeVenomHospitalService snakeVenomHospitalService;

    @Autowired
    private HospitalInfoService hospitalInfoService;

    // 原有接口保持不变

    /**
     * 检索周边支持蛇毒治疗的医院（核心扩展接口）
     * @param queryDTO 查询参数（用户输入、搜索半径、地图类型）
     * @return 筛选后的医院列表
     */
    @PostMapping("/snake-venom/nearby")
    public Result searchSnakeVenomHospitals(@Valid @RequestBody SnakeVenomHospitalQueryDTO queryDTO) {
        try {
            // 强制使用高德地图
            queryDTO.setMapType("AMAP");
            List<HospitalInfo> hospitals = snakeVenomHospitalService.searchSnakeVenomHospitals(queryDTO);
            if (hospitals.isEmpty()) {
                return ResultUtils.success("未找到支持蛇毒治疗的医院，建议扩大搜索半径", hospitals);
            }
            return ResultUtils.success("检索成功", hospitals);
        } catch (Exception e) {
            return ResultUtils.error("检索蛇毒治疗医院失败: " + e.getMessage());
        }
    }

    /**
     * 导出蛇毒治疗医院信息到Excel
     * @param queryDTO 查询参数
     * @return Excel文件下载响应
     */
    @PostMapping("/snake-venom/export")
    public ResponseEntity<byte[]> exportSnakeVenomHospitals(@Valid @RequestBody SnakeVenomHospitalQueryDTO queryDTO) {
        try {
            // 强制使用高德地图
            queryDTO.setMapType("AMAP");
            List<HospitalInfo> hospitals = snakeVenomHospitalService.searchSnakeVenomHospitals(queryDTO);
            byte[] excelBytes = snakeVenomHospitalService.exportSnakeVenomHospitalsToExcel(hospitals);

            // 设置响应头
            HttpHeaders headers = new HttpHeaders();
            headers.setContentType(MediaType.APPLICATION_OCTET_STREAM);
            headers.setContentDispositionFormData("attachment", "蛇毒治疗医院列表.xlsx");

            return new ResponseEntity<>(excelBytes, headers, HttpStatus.OK);
        } catch (Exception e) {
            return new ResponseEntity<>(HttpStatus.INTERNAL_SERVER_ERROR);
        }
    }

    /**
     * 生成医院导航链接（仅支持高德地图）
     * @param hospitalId 医院ID
     * @param userLng 用户经度
     * @param userLat 用户纬度
     * @return 高德地图导航链接
     */
    @GetMapping("/navigation")
    public Result generateNavigationUrl(@RequestParam Long hospitalId,
                                       @RequestParam Double userLng,
                                       @RequestParam Double userLat) {
        try {
            HospitalInfo hospital = hospitalInfoService.getById(hospitalId);
            if (hospital == null || hospital.getDelFlag()) {
                return ResultUtils.error("医院信息不存在");
            }

            double destLng = hospital.getLongitude().doubleValue();
            double destLat = hospital.getLatitude().doubleValue();

            // 高德地图导航链接（GCJ02坐标系）
            String navigationUrl = String.format(
                "https://uri.amap.com/navigation?from=%s,%s&to=%s,%s&mode=car&src=sling",
                userLng, userLat, destLng, destLat
            );

            return ResultUtils.success("导航链接生成成功", navigationUrl);
        } catch (Exception e) {
            return ResultUtils.error("生成导航链接失败: " + e.getMessage());
        }
    }
}

11. 核心扩展功能实现说明
11.1 蛇毒治疗医院检索流程
用户输入处理：支持模糊地址（如 "深圳市南山区"）或经纬度（如 "113.94,22.54"）两种输入方式位置解析：调用高德地图 API 将地址转换为 GCJ02 标准经纬度双重检索：
优先从本地数据库检索附近支持蛇毒治疗的医院（通过经纬度距离计算 + 关键词筛选）
若数据库无结果，调用高德地图 API 补充检索周边医院，并二次筛选蛇毒治疗相关资源
结果优化：
按距离从近到远排序，优先推荐最近的医疗资源
补充距离描述（如 "1.2 公里"）和匹配关键词，提升用户体验
支持 Excel 导出，方便离线使用
11.2 关键词筛选机制
采用多维度关键词匹配策略，避免遗漏有效医疗资源：
核心关键词库：覆盖 "蛇毒"" 抗蛇毒血清 ""蛇咬伤治疗" 等 10 + 核心术语
匹配维度：医院名称、蛇毒治疗关键词字段、医院类型标签、医院简介
匹配规则：任意关键词命中即视为支持蛇毒治疗的目标医院
11.3 坐标系处理
统一使用高德地图 GCJ02 坐标系（国家测绘局加密坐标系），无需进行坐标系转换，简化开发流程
12. 性能优化与扩展建议
12.1 性能优化
索引优化：对 snake_venom_treatment 和 snake_venom_keywords 字段建立索引，提升筛选效率
