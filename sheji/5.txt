步骤 2：豆包 API 调用工具类（封装核心逻辑）
java
运行
import com.alibaba.fastjson.JSON;
import com.alibaba.fastjson.JSONArray;
import com.alibaba.fastjson.JSONObject;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.http.HttpEntity;
import org.springframework.http.HttpHeaders;
import org.springframework.http.MediaType;
import org.springframework.http.ResponseEntity;
import org.springframework.stereotype.Component;
import org.springframework.web.client.RestTemplate;

import java.util.Collections;
import java.util.HashMap;
import java.util.Map;

@Component
public class DouBaoApiClient {
    @Value("${doubao.api.key}")
    private String apiKey;
    @Value("${doubao.api.url}")
    private String apiUrl;
    @Value("${doubao.api.model}")
    private String modelName;

    private final RestTemplate restTemplate;

    // 注入RestTemplate（建议配置超时时间）
    public DouBaoApiClient(RestTemplate restTemplate) {
        this.restTemplate = restTemplate;
    }

    /**
     * 调用豆包API获取指定蛇种的维基百科URL
     * @param snakeName 蛇种名称（如"眼镜蛇"）
     * @return 维基百科URL（如https://zh.wikipedia.org/zh/眼镜蛇）
     */
    public String getSnakeWikiUrl(String snakeName) {
        // 1. 构造精准Prompt（强制JSON输出，避免冗余）
        String prompt = String.format(
                "仅返回「%s」的维基百科官方网址，无需解释，输出格式为JSON：{\"wiki_url\":\"维基百科URL\"}",
                snakeName
        );

        // 2. 构造请求头（鉴权+数据格式）
        HttpHeaders headers = new HttpHeaders();
        headers.setContentType(MediaType.APPLICATION_JSON);
        headers.setBearerAuth(apiKey);  // 豆包API通常为Bearer Token鉴权（以官方文档为准）

        // 3. 构造请求体（匹配豆包API的参数规范）
        Map<String, Object> requestBody = new HashMap<>();
        requestBody.put("model", modelName);  // 你的模型名称
        requestBody.put("temperature", 0.0);  // 0=确定性输出，避免结果波动
        requestBody.put("max_tokens", 100);   // 限制响应长度，减少冗余
        requestBody.put("messages", Collections.singletonList(
                Map.of("role", "user", "content", prompt)
        ));

        // 4. 发送POST请求调用豆包API
        HttpEntity<Map<String, Object>> requestEntity = new HttpEntity<>(requestBody, headers);
        try {
            ResponseEntity<String> response = restTemplate.postForEntity(apiUrl, requestEntity, String.class);
            if (response.getStatusCode().is2xxSuccessful() && response.getBody() != null) {
                // 5. 解析豆包API响应（核心：匹配官方响应格式）
                JSONObject respJson = JSON.parseObject(response.getBody());
                JSONArray choices = respJson.getJSONArray("choices");
                if (choices != null && !choices.isEmpty()) {
                    String content = choices.getJSONObject(0).getJSONObject("message").getString("content");
                    // 解析大模型返回的JSON字符串，提取wiki_url
                    return JSON.parseObject(content).getString("wiki_url");
                }
            }
            throw new RuntimeException("豆包API响应无有效数据");
        } catch (Exception e) {
            throw new RuntimeException("调用豆包API失败：" + e.getMessage(), e);
        }
    }
}
步骤 3：维基百科图片获取（合规调用 MediaWiki API）
java
运行
import com.alibaba.fastjson.JSON;
import com.alibaba.fastjson.JSONObject;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.http.HttpHeaders;
import org.springframework.http.HttpEntity;
import org.springframework.http.ResponseEntity;
import org.springframework.stereotype.Component;
import org.springframework.web.client.RestTemplate;

import java.util.HashMap;
import java.util.Map;

@Component
public class WikiImageClient {
    @Value("${doubao.wiki.api-url}")
    private String wikiApiUrl;

    private final RestTemplate restTemplate;

    public WikiImageClient(RestTemplate restTemplate) {
        this.restTemplate = restTemplate;
    }

    /**
     * 从维基URL解析页面标题，调用MediaWiki API获取主图URL
     * @param wikiUrl 维基百科URL（如https://zh.wikipedia.org/zh/眼镜蛇）
     * @return 维基图片原始URL（如https://upload.wikimedia.org/wikipedia/commons/xxx.jpg）
     */
    public String getWikiImageUrl(String wikiUrl) {
        // 1. 解析维基URL中的页面标题（如"眼镜蛇"）
        String title = wikiUrl.substring(wikiUrl.lastIndexOf("/") + 1);
        // 处理URL编码（如"银环蛇"可能是%E9%93%B6%E7%8E%AF%E8%9B%87）
        try {
            title = java.net.URLDecoder.decode(title, "UTF-8");
        } catch (Exception e) {
            throw new RuntimeException("解析维基标题失败", e);
        }

        // 2. 构造维基API请求参数（参考MediaWiki官方文档）
        Map<String, String> params = new HashMap<>();
        params.put("action", "query");          // 核心动作：查询
        params.put("prop", "pageimages");       // 获取页面图片
        params.put("piprop", "original");       // 返回原图URL
        params.put("titles", title);            // 维基页面标题
        params.put("format", "json");           // 输出JSON格式
        params.put("formatversion", "2");       // 简化的JSON格式

        // 3. 构造请求头（维基要求必须设置User-Agent，否则会被封禁）
        HttpHeaders headers = new HttpHeaders();
        headers.add("User-Agent", "emergency-service/1.0 (你的联系邮箱)");  // 替换为真实信息

        // 4. 发送GET请求调用维基API
        HttpEntity<Map<String, String>> requestEntity = new HttpEntity<>(params, headers);
        try {
            ResponseEntity<String> response = restTemplate.postForEntity(wikiApiUrl, requestEntity, String.class);
            if (response.getStatusCode().is2xxSuccessful() && response.getBody() != null) {
                // 5. 解析维基API响应
                JSONObject respJson = JSON.parseObject(response.getBody());
                JSONObject pages = respJson.getJSONObject("query").getJSONArray("pages").getJSONObject(0);
                if (pages.containsKey("original")) {
                    return pages.getJSONObject("original").getString("source");  // 图片原始URL
                }
                throw new RuntimeException("维基页面无主图");
            }
            throw new RuntimeException("维基API响应异常");
        } catch (Exception e) {
            throw new RuntimeException("获取维基图片失败：" + e.getMessage(), e);
        }
    }
}
步骤 4：emergency-service 业务接口（对外提供服务）
java
运行
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;

@RestController
@RequestMapping("/emergency-service/snake")
public class SnakeEmergencyController {
    private final DouBaoApiClient douBaoApiClient;
    private final WikiImageClient wikiImageClient;

    public SnakeEmergencyController(DouBaoApiClient douBaoApiClient, WikiImageClient wikiImageClient) {
        this.douBaoApiClient = douBaoApiClient;
        this.wikiImageClient = wikiImageClient;
    }

    /**
     * 获取指定蛇种的维基百科信息（URL+图片）
     * @param snakeName 蛇种名称（如"眼镜蛇"）
     * @return 包含URL和图片的JSON
     */
    @GetMapping("/wiki-info")
    public Map<String, String> getSnakeWikiInfo(@RequestParam String snakeName) {
        Map<String, String> result = new HashMap<>();
        // 1. 调用豆包API获取维基URL
        String wikiUrl = douBaoApiClient.getSnakeWikiUrl(snakeName);
        result.put("wiki_url", wikiUrl);
        // 2. 调用维基API获取图片URL
        String imageUrl = wikiImageClient.getWikiImageUrl(wikiUrl);
        result.put("image_url", imageUrl);
        return result;
    }
}
步骤 5：RestTemplate 配置（全局 Bean）
java
运行
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.web.client.RestTemplate;

@Configuration
public class RestTemplateConfig {
    @Bean
    public RestTemplate restTemplate() {
        RestTemplate restTemplate = new RestTemplate();
        // 配置超时时间（可选，避免接口阻塞）
        restTemplate.setRequestFactory(new org.springframework.http.client.SimpleClientHttpRequestFactory() {{
            setConnectTimeout(5000);  // 连接超时5s
            setReadTimeout(10000);   // 读取超时10s
        }});
        return restTemplate;
    }
}

四、关键注意事项
API 鉴权验证：若豆包 API 的鉴权方式不是Bearer Token，需对照官方文档调整请求头（如api-key直接放在 Header 中）；
跨域问题：维基图片 URL 可能存在跨域限制，需在emergency-service中增加「图片代理接口」（前端请求微服务，微服务转发图片流）：
java
运行
@GetMapping("/proxy-image")
public ResponseEntity<byte[]> proxyImage(@RequestParam String imageUrl) {
    RestTemplate restTemplate = new RestTemplate();
    byte[] imageBytes = restTemplate.getForObject(imageUrl, byte[].class);
    HttpHeaders headers = new HttpHeaders();
    headers.setContentType(MediaType.IMAGE_JPEG);  // 根据实际图片类型调整
    return new ResponseEntity<>(imageBytes, headers, HttpStatus.OK);
}
前端调用时将image_url替换为/emergency-service/snake/proxy-image?image_url=xxx；
容错处理：增加空值判断（如维基无图片、豆包返回非预期格式），避免前端报错；
合规性：维基图片仅限非商用场景，且需保留维基的版权声明；
API 限流：豆包 API 可能有 QPS 限制，建议在微服务中增加限流（如 Guava RateLimiter）。
五、调试排查要点
先通过 Postman 调用豆包 API（直接请求https://api.doubao.com/v1/chat/completions），确认能返回正确的维基 URL；
检查维基 API 的 User-Agent 是否配置，无 User-Agent 会被维基封禁；
若图片无法加载，优先排查跨域 / 代理问题，或维基图片 URL 是否有效。
